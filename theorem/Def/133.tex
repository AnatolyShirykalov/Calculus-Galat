 	Непустое семейство $\B$ подмножеств $X$ называют базой (фильтром, базис-фильтром, но у нас будет непафосно: база), если
 	\begin{enumerate}
 	  \item $\q\nin \B$;
 	  \item $\forall\  B_1,B_2\in\B\pau \exists\  B_3\in\B\colon B_3\subset B_1\cap B_2$.
 	\end{enumerate}
